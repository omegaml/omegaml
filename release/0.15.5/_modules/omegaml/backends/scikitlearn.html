 32    def _v1_package_model(self, model, filename):
 33        """
 34        Dumps a model using joblib and packages all of joblib files into a zip
 35        file
 36        """
 37        import joblib
 38        lpath = tempfile.mkdtemp()
 39        fname = os.path.basename(filename)
 40        mklfname = os.path.join(lpath, fname)
 41        zipfname = os.path.join(self.model_store.tmppath, fname)
 42        joblib.dump(model, mklfname, protocol=4)
 43        with ZipFile(zipfname, 'w', compression=ZIP_DEFLATED) as zipf:
 44            for part in glob.glob(os.path.join(lpath, '*')):
 45                zipf.write(part, os.path.basename(part))
 46        rmtree(lpath)
 47        return zipfname

 49    def _v1_extract_model(self, packagefname):
 50        """
 51        Loads a model using joblib from a zip file created with _package_model
 52        """
 53        import joblib
 54        lpath = tempfile.mkdtemp()
 55        fname = os.path.basename(packagefname)
 56        mklfname = os.path.join(lpath, fname)
 57        with ZipFile(packagefname) as zipf:
 58            zipf.extractall(lpath)
 59        model = joblib.load(mklfname)
 60        rmtree(lpath)
 61        return model

 63    def _v1_get_model(self, name, version=-1):
 64        """
 65        Retrieves a pre-stored model
 66        """
 67        filename = self.model_store._get_obj_store_key(name, '.omm')
 68        packagefname = os.path.join(self.model_store.tmppath, name)
 69        dirname = os.path.dirname(packagefname)
 70        try:
 71            os.makedirs(dirname)
 72        except OSError:
 73            # OSError is raised if path exists already
 74            pass
 75        meta = self.model_store.metadata(name, version=version)
 76        outf = self.model_store.fs.get_version(filename, version=version)
 77        with open(packagefname, 'wb') as zipf:
 78            zipf.write(meta.gridfile.read())
 79        model = self._v1_extract_model(packagefname)
 80        return model

 82    def _v1_put_model(self, obj, name, attributes=None, **kwargs):
 83        """
 84        Packages a model using joblib and stores in GridFS
 85        """
 86        zipfname = self._v1_package_model(obj, name)
 87        with open(zipfname, 'rb') as fzip:
 88            gridfile = self.model_store.fs.put(
 89                fzip, filename=self.model_store._get_obj_store_key(name, 'omm'))
 90        return self.model_store._make_metadata(
 91            name=name,
 92            prefix=self.model_store.prefix,
 93            bucket=self.model_store.bucket,
 94            kind=MDREGISTRY.SKLEARN_JOBLIB,
 95            attributes=attributes,
 96            gridfile=gridfile).save()

285            'Xname': Xname,
286            'Yname': Yname,
287            'gsModel': gs_modelname,
288        })

290        return meta

297def process(it, fn=None, keep=False, keep_last=False):
298    """
299    consume iterator, optionally keeping results and calling a function
300
301    Args:
302        it: iterator to consume
303        fn: optional, a function to call on result as result = fn(result)
304        keep: optional, if True returns all results, False by default
305        keep_last: optional, if True returns last result, False by default
306
307    Returns:
308        list of results if keep is True
309        last result if keep_last is True
310        None if keep and keep_last are False
311    """
312    results = [] if keep else None
313    result = None
314    for result in it:
315        result = fn(result) if fn else result
316        results.append(result) if keep else None
317    return result if keep_last else results

320def maybe_chunked(meth, argfn, *args, **kwargs):
321    """call a function with given args, kwargs, perhaps iterating over args
322
323    Calls meth(*args, **kwargs) by the following rules:
324
325    * if all values in *args are generators call meth() until generators exhausted
326    * else run meth() once using *args as a single chunk
327    * before each meth() call, argfn(*chunks) is called
328
329    If all values in *args are generators this is equivalent to
330
331        for values in zip(*args):
332            yield meth(*values, **kwargs)
333
334    If values in *args are not generators this is equivalent to
335
336        yield meth(*args, **kwargs)
337
338    Usage:
339        X = get(...)
340        Y = get(...)
341
342        # original call
343        model.partial_fit(X, Y, **kwargs)
344
345        # transformed call, assuming X, Y can be generators over partial data
346        # this will work whether X, Y are generators or not.
347        for result in maybe_chunked(model.partial_fit,
348                                    lambda *args: as_args(*args),
349                                    X, Y, **kwargs):
350            ... # process results as needed
351
352        Note the (lambda ...), which is called on the X, Y args for each call
353        on model.partial_fit. Use this to apply functions to each chunk, e.g.:
354
355            # here the lambda receives the values of X, Y in each iteration
356            maybe_chunked(...,
357                          lambda X, Y: as_args(reshaped(X), reshaped(Y)),
358                          ...)
359
360        # combine with process() to simplify processing
361        # essentiall this implements a for loop around maybe_chunked()
362        resultl = process(maybe_chuncked(...), keep_last=True)
363
364    Rationale:
365
366        Avoid repeated code of the form
367
368            X, Y = generators or actual values
369            if isgenerator(X) and isgenerator(Y):
370                for Xc, Yc in zip(X, Y):
371                    chunk_result = meth(reshape(Xc), reshape(Yc), **kwargs)
372                    # add logic to process chunk_result if needed
373                    ...
374                result = combine chunk_results, or keep only last
375            else:
376                result = meth(reshape(X), reshape(Y), **kwargs)
377
378        Instead this can be written as
379
380            result = process(maybe_chunked(meth,
381                                           lambda X, Y: as_args(reshape(X), reshape(Y)),
382                                           X, Y, **kwargs),
383                             keep_last=True)
384
385    Args:
386        meth: function to call
387        argfn: function to call on values for each iteration
388        *args: arguments to meth
389        **kwargs: kwargs to meth
390
391    Returns:
392        generator of meth() results
393    """
394    isgenerator = lambda v: isinstance(v, types.GeneratorType)
395    if all(isgenerator(v) for v in args):
396        for chunks in zip(*args):
397            yield meth(*argfn(*chunks), **kwargs)
398    else:
399        yield meth(*argfn(*args), **kwargs)

402def as_args(*args):
403    # return the arguments as passed in
404    # this is to enable e.g. lambda X: as_args(X), where X is an iterable
405    # notice if we did not use as_args a subsequent fn(*(lambda X: X)) would
406    # not be the same as fn(X), however fn(*(lambda X: as_args(X)) is equiv.
407    return args
