 43    def _parse_url(self, obj):
 44        # properly parse RFC 1808 URLs
 45        # -- format: <vendor>+<protocol>://<base_url>;<params>?<query>
 46        # -- https://datatracker.ietf.org/doc/html/rfc1808#section-2.1
 47        # Python urlparse does not support ;params parsing in custom schemes
 48        # -- depending on the path, ;params ends up in path, netloc or hostname
 49        # -- we fix that by adjusting the parsed result
 50        # -- .vendor is the vendor, e.g. openai
 51        # -- .scheme is the protocol, e.g. https
 52        parsed = urlparse(obj)
 53        vendor, scheme = parsed.scheme.split('+') if '+' in parsed.scheme else ('', 'https')
 54        path, params = parsed.path.split(';', 1) if ';' in parsed.path else (parsed.path, '')
 55        netloc, params = parsed.netloc.split(';', 1) if ';' in parsed.netloc else (parsed.netloc, params)
 56        hostname, params = parsed.hostname.split(';', 1) if ';' in parsed.hostname else (parsed.hostname, params)
 57        port = parsed.port or (443 if scheme == 'https' else 80)
 58        ParseResult = namedtuple('ParseResult', ['vendor', 'scheme',
 59                                                 'path', 'params', 'netloc', 'hostname', 'port', 'username', 'password',
 60                                                 'query'])
 61        return ParseResult(vendor, scheme, path, params, netloc, hostname, port, parsed.username,
 62                           parsed.password, parsed.query)

 64    def put(self, obj, name, template=None, pipeline=None, provider=None, tools=None, documents=None, **kwargs):
 65        self.model_store: OmegaStore
 66        parsed = self._parse_url(obj)
 67        params = parse_qs(parsed.params)
 68        if 'model' in params:
 69            model = params.pop('model')[0]
 70            path = parsed.path
 71        else:
 72            path, model = parsed.path.split('/', 1) if '/' in parsed.path else (parsed.path, None)
 73        assert model, f'no model specified in {obj}, use openai://<base_url>;model=<model> or openai+<scheme>://<base_url>;model=<model>'
 74        base_url = f'{parsed.scheme}://{parsed.hostname}:{parsed.port}{path}'
 75        creds = f'{parsed.username}:{parsed.password}' if parsed.username and parsed.password else ''
 76        creds = creds or f'{parsed.username}' if parsed.username else ''
 77        query = parse_qs(parsed.query)
 78        params = params or {}
 79        query = query or {}
 80        provider = provider or self._infer_provider(base_url)
 81        kind_meta = {
 82            'base_url': base_url,
 83            'creds': creds,
 84            'model': model,
 85            'query': query,
 86            'params': params,
 87            'provider': provider,
 88        }
 89        attributes = {
 90            'template': template or query.get('template'),
 91            'dataset': self._messages_dataset(name),
 92            'pipeline': pipeline or None,
 93            'tools': tools or [],
 94            'documents': documents or [],
 95        }
 96        kwargs.update(attributes=attributes)
 97        meta = self.model_store.make_metadata(name,
 98                                              kind=self.KIND,
 99                                              kind_meta=kind_meta,
100                                              **kwargs)
101        return meta.save()

105        kind_meta = meta.kind_meta

125                          dataset=self._messages_dataset(name, meta=meta),
126                          provider=provider, documents=documents,

163class TextModel(GenAIModel):
164    """ OpenAI model
165
166    This implements the OpenAI model interface. It is a thin wrapper around the OpenAI API,
167    and adds conversation tracking and data storage for the conversation history. For chat completions,
168    the conversation history is stored in a dataset named ./openai/messages/<modelname>/<user>. For
169    completions without a conversation id, a new conversation id is generated and returned in each
170    message, however the conversation history is not stored in this case. The complete() method
171    can be called with a conversation id to continue a conversation, in this case it is equivalent
172    to chat().
173
174    The model implements a callback to a user function or virtul object handler, called the pipeline.
175    The pipeline is called with the method name, the conversation id, the prompt message, and
176    the messages so far. It can be used to implement custom logic for preparing the messages and
177    the response.
178
179    Usage:
180
181        Create and access a model::
182
183            # create a model
184            om.models.put('openai://localhost:8000/mymodel', 'mymodel')
185            model = om.models.get('mymodel')
186            # complete a prompt
187            result = model.complete('hello, how are you?')
188            # chat
189            conversation_id, result = model.chat('hello, how are you?')
190            # continue a conversation
191            result = model.complete('I am fine, thank you.', conversation_id=conversation_id)
192
193        Implement a pipeline::
194
195            # add a pipeline
196            @virtual_genai
197            def my_pipeline(method, conversation_id, prompt_message, messages):
198                # implement your logic here
199                if method == 'prepare':
200                    # prepare the initial messages
201                    return messages
202                elif method == 'template':
203                    # prepare the template
204                    return 'You are a helpful assistant.'
205                elif method == 'process':
206                    # process the response
207                    return response_message
208
209            model = om.models.get('mymodel', pipeline=my_pipeline)
210            result = model.complete('hello, how are you?')
211
212            # store the pipeline in a virtual object
213            om.models.put(my_pipeline, 'my_pipeline')
214            model = om.models.put('openai://localhost:8000/mymodel', 'mymodel', pipeline='my_pipeline')
215            # this will automatically load the pipeline, and get it called for each stage
216            result = model.complete('hello, how are you?')
217
218        A pipeline can be used to implement custom logic for preparing the messages and to check
219        or change the response. The pipeline can return any messages, a custom template, or
220        a custom response.
221
222        Get back the conversation history::
223
224            model = om.models.get('mymodel')
225            messages = model.conversation(conversation_id)
226    """

264            conversation_id, response, _, response_message = r

267        if not chat and conversation_id is None:

280        # return response(s)
281        response_gen = (response_parser(response) for response in responses)

298    def _do_chat(self, prompt, conversation_id=None, data=None, use_tools=False, raw=False, **kwargs):
299        assert self.data_store, "chat requires a data_store, specify data_store=om.datasets"
300        conversation_id = conversation_id or uuid4().hex
301        messages = self.conversation(conversation_id)
302        if hasattr(messages, 'to_dict'):
303            # convert pandas dataframe to records
304            messages = messages.to_dict('records')
305        if messages is None:
306            messages = [self._system_message(prompt, conversation_id=conversation_id)]
307            self.data_store.put(messages, self.dataset, kind='pandas.rawdict')
308        responses = self._do_complete(prompt, messages, conversation_id=conversation_id, data=data,
309                                      use_tools=use_tools, raw=raw, **kwargs)
310        to_store = []
311        for response in responses:
312            response, prompt_message, response_message = response
313            # wrapping in dict() to avoid modification by the data store (e.g. adding _id)
314            to_store = [
315                dict(prompt_message),
316                dict(response_message)
317            ]
318            yield conversation_id, response, prompt_message, response_message
319        else:
320            self.data_store.put(to_store, self.dataset, kind='pandas.rawdict')

322    def _call_tools(self, tool_calls, conversation_id):
323        # process tool calls
324        results = []
325        tool_prompts = []
326        for tool_call in tool_calls:
327            tool = [(ts, tf) for ts, tf in zip(self.tools_specs, self.tools)
328                    if tf.__name__ == tool_call['function']['name']]
329            if tool:
330                tool, tool_func = tool[0]
331                tool_kwargs = json.loads(tool_call['function']['arguments'])
332                try:
333                    tool_result = tool_func(**tool_kwargs)
334                except Exception as e:
335                    tool_result = str(e)
336                tool_response = {
337                    "role": "assistant",
338                    "content": tool_result,
339                    "conversation_id": conversation_id,
340                }
341                results.append(tool_response)
342                tool_prompts.append({
343                    "role": "tool",
344                    "tool_call_id": tool_call["id"],
345                    "content": str(tool_result)
346                })
347        return results, tool_prompts

366    def _do_complete(self, prompt, messages=None, conversation_id=None, data=None, stream=False,
367                     use_tools=False, raw=False, **kwargs):
368        conversation_id = conversation_id or uuid4().hex
369        messages = messages or []
370        if prompt and isinstance(prompt, str):
371            # support direct text input
372            prompt_message = {
373                "role": "user",
374                "content": self._augment_prompt(prompt, self.documents),
375                "conversation_id": conversation_id,
376            }
377            messages = ([self._system_message(prompt, conversation_id=conversation_id)] +
378                        [self._augment_message(m, self.documents) for m in messages])
379        elif isinstance(prompt, dict):
380            # support structured input
381            # -- see OpenAI /chat/completions endpoint, "messages" parameter
382            #    https://platform.openai.com/docs/api-reference/chat
383            # -- assume prompt is a fully formed provider-compatible message,e.g. from a chat client
384            messages = ([self._system_message(prompt.get('content', ''), conversation_id=conversation_id)] +
385                        [self._augment_message(m, self.documents) for m in messages])
386            prompt_message = prompt
387        elif isinstance(prompt, list):
388            # support structured input, as messages
389            # -- see OpenAI /chat/completions endpoint, "messages" parameter
390            #    https://platform.openai.com/docs/api-reference/chat
391            # -- assume prompt is a fully formed provider-compatible message,e.g. from a chat client
392            prompts = '\n\n'.join(m.get('content', '') for m in prompt)
393            messages = ([self._system_message(prompts,
394                                              conversation_id=conversation_id)] +
395                        [self._augment_message(m, self.documents) for m in prompt])
396            prompt_message = None
397        else:
398            # raw input, assume messages contains the user prompt
399            prompt_message = None
400            prompts = '\n\n'.join(m.get('content', '') for m in messages)
401            messages = ([self._system_message(prompts,
402                                              conversation_id=conversation_id)] +
403                        [self._augment_message(m, self.documents) for m in messages])
404        if self.tools:
405            kwargs.update(tools=self.tools_specs,
406                          tool_choice='auto')
407        _template = self._prepare_template(self.template,
408                                           data=data)
409        template = self.pipeline(method='template',
410                                 prompt_message=prompt_message,
411                                 messages=messages,
412                                 template=self.template,
413                                 conversation_id=conversation_id,
414                                 **kwargs) or _template
415        _default_messages = (messages + [prompt_message]) if prompt_message else messages
416        messages = self.pipeline(method='prepare',
417                                 prompt_message=prompt_message,
418                                 messages=messages,
419                                 template=template,
420                                 conversation_id=conversation_id, **kwargs) or _default_messages
421        response = self.provider.complete(
422            messages=messages,
423            stream=stream,
424            model=self.model,
425            **kwargs
426        )
427
428        def maybe_call_tools(response, prompt_message, response_message, use_tools=False, as_delta=False):
429            """ prepare calling tools, optionally actually call the selected tool """
430            if hasattr(response.choices[0], 'delta'):
431                message = response.choices[0].delta
432            else:
433                message = response.choices[0].message
434            if self.tools and getattr(message, 'tool_calls', None):
435                # call tools
436                response_message['tool_calls'] = message.to_dict()['tool_calls']
437                tool_calls = [tool.to_dict() for tool in message.tool_calls]
438            else:
439                tool_calls = None
440            if use_tools and tool_calls:
441                results, tool_prompts = self._call_tools(tool_calls, conversation_id)
442                # ask llm to respond to tool results
443                # -- avoid recursive tool calls
444                # -- never stream results
445                kkwargs = dict(kwargs)
446                kkwargs.pop('stream', None)
447                kkwargs.pop('tools', None)
448                kkwargs.pop('tool_choice', None)
449                toolcall_messages = messages + [prompt_message] + [message] + tool_prompts
450                toolcall_messages = self.pipeline(method='toolcall',
451                                                  prompt_message=prompt_message,
452                                                  messages=toolcall_messages,
453                                                  template=template,
454                                                  conversation_id=conversation_id, **kwargs) or toolcall_messages
455                response = self.provider.complete(
456                    messages=toolcall_messages,
457                    model=self.model,
458                    **kkwargs,
459                )
460                tooled_response, tooled_prompt_message, tooled_response_message = resolve_response(response,
461                                                                                                   prompt_message,
462                                                                                                   use_tools=False)
463                tooled_response_message['intermediate_results'] = {
464                    'tool_calls': tool_calls,
465                    'tool_prompts': tool_prompts,
466                    'tool_results': results,
467                }
468                tooled_response_message = self.pipeline(method='toolresult', response_message=tooled_response_message,
469                                                        prompt_message=prompt_message,
470                                                        messages=toolcall_messages,
471                                                        template=template,
472                                                        conversation_id=conversation_id,
473                                                        **kwargs) or tooled_response_message
474                if as_delta:
475                    # ensure the tool response is shown as a delta chunk
476                    if raw:
477                        tooled_response_message['choices'][0]['delta'] = tooled_response_message['choices'][0].pop(
478                            'message', None)
479                    else:
480                        tooled_response_message['delta'] = response_message['content']
481                return tooled_response, prompt_message, tooled_response_message
482            return response, prompt_message, response_message
483
484        def resolve_response(response, prompt_message, use_tools=False):
485            if raw:
486                # native response message
487                # Ref: https://platform.openai.com/docs/api-reference/chat/get
488                response_message = response.to_dict()
489            elif getattr(response, 'error', None):
490                return response, prompt_message, {
491                    "role": "system",
492                    "content": response.error.get('message', str(response.error)),
493                    "conversation_id": conversation_id,
494                    "error": response.error,
495                }
496            else:
497                response_message = {
498                    "role": response.choices[0].message.role,
499                    "content": tryOr(lambda: response.choices[0].message.content, None),
500                    "conversation_id": conversation_id,
501                }
502            response, prompt_message, response_message = maybe_call_tools(response, prompt_message,
503                                                                          response_message, use_tools=use_tools)
504            response_message = self.pipeline(method='process', response_message=response_message,
505                                             prompt_message=prompt_message,
506                                             messages=messages,
507                                             template=template,
508                                             conversation_id=conversation_id,
509                                             **kwargs) or response_message
510            return response, prompt_message, response_message
511
512        def resolve_chunk(response, chunk, chunks, prompt_message, use_tools=False):
513            if raw:
514                response_message = chunk.to_dict()
515            else:
516                response_message = {
517                    "role": chunk.choices[0].delta.role,
518                    "delta": chunk.choices[0].delta.content,
519                    "content": ''.join(c['delta'] for c in chunks) + str(chunk.choices[0].delta.content or ''),
520                    "conversation_id": conversation_id,
521                }
522            response, prompt_message, response_message = maybe_call_tools(chunk, prompt_message,
523                                                                          response_message, use_tools=use_tools,
524                                                                          as_delta=True)
525            chunks.append(response_message)
526            response_message = self.pipeline(method='process', response_message=response_message,
527                                             prompt_message=prompt_message,
528                                             messages=messages,
529                                             template=template,
530                                             conversation_id=conversation_id,
531                                             **kwargs) or response_message
532            return response, prompt_message, response_message
533
534        if stream:
535            # https://cookbook.openai.com/examples/how_to_stream_completions
536            print("*** streaming")
537            chunks = []
538            for chunk in response:
539                yield resolve_chunk(response, chunk, chunks, prompt_message, use_tools=use_tools)
540        else:
541            yield resolve_response(response, prompt_message, use_tools=use_tools)

550    def _get_function_spec(self, func):
551        """
552        Generates an OpenAI SDK function dictionary from an annotated function.
553
554        Args:
555            func (callable): The annotated function to be converted.
556
557        Returns:
558            dict: The OpenAI SDK function dictionary.
559
560        References:
561            - https://platform.openai.com/docs/api-reference/debugging-requests
562        """
563        if isinstance(func, dict):
564            return func
565        import inspect
566        sig = inspect.signature(func)
567        params = {}
568        TYPES = {
569            str: 'string',
570            int: 'integer',
571            float: 'float',
572            list: 'list',
573        }
574        for param in sig.parameters.values():
575            param_type = TYPES.get(
576                param.annotation) or 'string' if param.annotation != inspect.Parameter.empty else None
577            param_default = param.default if param.default != inspect.Parameter.empty else None
578            param_type = param_type or TYPES.get(type(param_default))
579            param_dict = {
580                "name": param.name,
581                "type": param_type,
582                "description": f'represents {param.name}',
583                "default": param_default,
584            }
585            params[param.name] = param_dict
586
587        return_type = str(sig.return_annotation) if sig.return_annotation != inspect.Parameter.empty else None
588
589        function_dict = {
590            "type": "function",
591            "function": {
592                "name": func.__name__,
593                "description": inspect.getdoc(func) or 'A function to return a response',
594                "parameters": {
595                    "type": "object",
596                    "properties": {
597                        k: v for k, v in params.items()
598                    },
599                },
600                "return_type": return_type,
601            }
602        }
603
604        return function_dict
605
606    def _augment_prompt(self, prompt, documents: DocumentIndex, query=None):
607        query = query or prompt
608        if documents is None or '{context}' not in prompt:
609            return prompt
610        docs = documents.retrieve(query)
611        if docs:
612            context = ensure_list(docs)[0].get('text')
613        else:
614            context = '(error: no documents found)'
615        return prompt.format_map(safeformat(context=context))
616
617    def _augment_message(self, message, documents: DocumentIndex, query=None):
618        augmented = self._augment_prompt(message.get('content', ''), documents=documents, query=query)
619        message['content'] = augmented if augmented else message.get('content')
620        return message

645    def __init__(self, *args, **kwargs):
646        super().__init__(*args, **kwargs)

666            stream=stream,

672class JinaEmbeddingsProvider(Provider):
673    URL_REGEX = r'https?://(api\.jina\.ai)(:\d+)?/.*'
674
675    def embed(self, documents, dimensions=None, model=None, **kwargs):
676        """ Embed documents using Jina AI's embedding service.
677
678        Args:
679            documents (list): List of documents to embed.
680            dimensions (int): Number of dimensions to embed to.
681            model (str): Model name to use for embedding.
682
683        Returns:
684            list: List of embeddings as list[list[float, ...]].
685        """
686        # see https://jina.ai/embeddings
687        documents = ensure_list(documents)
688        headers = {
689            'Content-Type': 'application/json',
690            'Authorization': f'Bearer {self.api_key}'
691        }
692        url = urljoin(self.base_url, 'embeddings')
693        resp = requests.post(url,
694                             headers=headers,
695                             json={'model': self.model,
696                                   'input': [{
697                                       'text': doc
698                                   } for doc in documents]})
699        assert resp.status_code == 200, f'Error {resp.status_code} calling {url}: {resp.text}'
700        data = resp.json().get('data', [])
701        return [d['embedding'] for d in sorted(data, key=lambda x: x['index'])]
