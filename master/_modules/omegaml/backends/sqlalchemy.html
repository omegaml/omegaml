 39class SQLAlchemyBackend(BaseDataBackend):
 40    """
 41    sqlalchemy plugin for omegaml
 42
 43    Usage:
 44
 45        Define your sqlalchemy connection::
 46
 47            sqlalchemy_constr = f'sqlalchemy://{user}:{password}@{account}/'
 48
 49        Store the connection in any of three ways::
 50
 51            # -- just the connection
 52            om.datasets.put(sqlalchemy_constr, 'mysqlalchemy')
 53            om.datasets.get('mysqlalchemy', raw=True)
 54            => the sql connection object
 55
 56            # -- store connection with a predefined sql
 57            om.datasets.put(sqlalchemy_constr, 'mysqlalchemy',
 58                            sql='select ....')
 59            om.datasets.get('mysqlalchemy')
 60            => will return a pandas dataframe using the specified sql to run.
 61               specify chunksize= to return an interable of dataframes
 62
 63            # -- predefined sqls can contain variables to be resolved at access time
 64            #    if you miss to specify required variables in sqlvars, a KeyError is raised
 65            om.datasets.put(sqlaclhemy_constr, 'myview',
 66                            sql='select ... from T1 where col="{var}"')
 67            om.datasets.get('mysqlalchemy', sqlvars=dict(var="value"))
 68
 69            # -- Variables are replaced by binding parameters, which is safe for
 70            #    untrusted inputs. To replace variables as strings, use double
 71            #    `{{variable}}` notation. A warning will be issued because this
 72            #    is considered an unsafe practice for untrusted input (REST API).
 73            #    It is in your responsibility to sanitize the value of the `cols` variable.
 74            om.datasets.put(sqlaclhemy_constr, 'myview',
 75                            sql='select {{cols}} from T1 where col="{var}"')
 76            om.datasets.get('mysqlalchemy', sqlvars=dict(cols='foo, bar',
 77                                                         var="value"))
 78
 79        Query data from a connection and store into an omega-ml dataset::
 80
 81            # -- copy the result of the sqlalchemy query to omegaml
 82            om.datasets.put(sqlalchemy_constr, 'mysqlalchemy',
 83                            sql='select ...', copy=True)
 84            om.datasets.get('mysqlalchemy')
 85            => will return a pandas dataframe (without executing any additional queries)
 86            => can also use with om.datasets.getl('mysqlalchemy') to return a MDataFrame
 87
 88        Controlling the table used in the connection::
 89
 90            # -- the default table is {bucket}_{name}, override using table='myname'
 91            om.datasets.put(sqlalchemy_constr, 'mysqlalchemy',
 92                            table='mytable',
 93                            sql='select ...',
 94                            copy=True)
 95            om.datasets.get('mysqlalchemy') # read from {bucket}_myname
 96
 97            # -- to use a specific table, without bucket information use table=':myname'
 98            om.datasets.put(sqlalchemy_constr, 'mysqlalchemy',
 99                            table=':mytable',
100                            sql='select ...',
101                            copy=True)
102            om.datasets.get('mysqlalchemy') # read from myname
103
104        Inserting data via a previously stored connection::
105
106            # -- store data back through the connection
107            om.datasets.put(sqlalchemy_constr, 'mysqlalchemy')
108            om.datasets.put(df, 'mysqlalchemy',
109                            table='SOMETABLE')
110
111        Using variables in connection strings:
112
113        Connection strings may contain variables, e.g. userid and password.
114        By default variables are resolved from the os environment. Can also
115        specify using any dict.::
116
117            # -- use connection string with variables
118            sqlalchemy_constr = 'sqlite:///{dbname}.db'
119            om.datasets.put(sqlalchemy_constr, 'userdb')
120            om.datasets.get('userdb', secrets=dict(dbname='chuckdb'))
121
122            # -- alternatively, create a vault dataset:
123            secrets = dict(userid='chuck', dbname='chuckdb')
124            om.datasets.put(secrets, '.omega/vault')
125            om.datasets.get('userdb')
126
127            the '.omega/vault' dataset will be queried using the current userid as
128            the secret name, and the dbname retrieved from the document. This is
129            experimental and the vault is not encrypted.
130
131    Advanced:
132
133        ``om.datasets.put()`` supports the following additional keyword arguments
134
135        * ``chunksize=int`` - specify the number of rows to read from sqlalchemy in one chunk.
136          defaults to 10000
137
138        * ``parse_dates=['col', ...]`` - list of column names to parse for date, time or datetime.
139          see pd.read_sql for details
140
141        * ``transform=callable`` - a callable, is passed the DataFrame of each chunk before it
142          is inserted into the database. use to provide custom transformations.
143          only works on copy=True
144
145        * any other kwargs supported by ``pandas.read_sql``

181    def get(self, name, sql=None, chunksize=None, raw=False, sqlvars=None,
182            secrets=None, index=True, keep=None, lazy=False, table=None, trusted=False, *args, **kwargs):
183        """ retrieve a stored connection or query data from connection
184
185        Args:
186            name (str): the name of the connection
187            secrets (dict): dict to resolve variables in the connection string
188            keep (bool): if True connection is kept open, defaults to True (change
189              default by setting om.defaults.SQLALCHEMY_ALWAYS_CACHE = False)
190            table (str): the name of the table, will be prefixed with the
191               store's bucket name unless the table is specified as ':name'
192            trusted (bool|str): if passed must be the value for store.sign(sqlvars or kwargs),
193               otherwise a warning is issued for any remaining variables in the sql statement
194
195        Returns:
196            connection
197
198        To query data and return a DataFrame, specify ``sql='select ...'``:
199
200        Args:
201                sql (str): the sql query, defaults to the query specific on .put()
202                chunksize (int): the number of records for each chunk, if
203                   specified returns an iterator
204                sqlvars (dict): optional, if specified will be used to format sql
205
206        Returns:
207            pd.DataFrame
208
209        To get the connection for a data query, instead of a DataFrame:
210
211        Args:
212
213                raw (bool): if True, returns the raw sql alchemy connection
214                keep (bool): option, if True keeps the connection open. Lazy=True
215                  implies keep=True. This is potentially unsafe in a multi-user
216                  environment where connection strings contain user-specific
217                  secrets. To always keep connections open, set
218                  ``om.datasets.defaults.SQLALCHEMY_ALWAYS_CACHE=True``
219
220        Returns:
221            connection
222
223        To get a cursor for a data query, instead of a DataFrame. Note this
224        implies keep=True.
225
226        Args:
227
228                lazy (bool): if True, returns a cursor instead of a DataFrame
229                sql (str): the sql query, defaults to the query specific on .put()
230
231        Returns:
232            cursor
233        """
234        meta = self.data_store.metadata(name)
235        connection_str = meta.kind_meta.get('sqlalchemy_connection')
236        valid_sql = lambda v: isinstance(v, str) or v is not None
237        sql = sql if valid_sql(sql) else meta.kind_meta.get('sql')
238        sqlvars = sqlvars or {}
239        table = self._default_table(table or meta.kind_meta.get('table') or name)
240        if not raw and not valid_sql(sql):
241            sql = f'select * from :sqltable'
242        chunksize = chunksize or meta.kind_meta.get('chunksize')
243        _default_keep = getattr(self.data_store.defaults,
244                                'SQLALCHEMY_ALWAYS_CACHE',
245                                ALWAYS_CACHE)
246        keep = keep if keep is not None else _default_keep
247        if connection_str:
248            secrets = self._get_secrets(meta, secrets)
249            connection = self._get_connection(name, connection_str, secrets=secrets, keep=keep)
250        else:
251            raise ValueError('no connection string')
252        if not raw and valid_sql(sql):
253            sql = sql.replace(':sqltable', table)
254            index_cols = _meta_to_indexcols(meta) if index else kwargs.get('index_col')
255            stmt = self._sanitize_statement(sql, sqlvars, trusted=trusted)
256            kwargs = meta.kind_meta.get('kwargs') or {}
257            kwargs.update(kwargs)
258            if not lazy:
259                logger.debug(f'executing sql {stmt} with parameters {sqlvars}')
260                pd_kwargs = {**dict(chunksize=chunksize, index_col=index_cols,
261                                    params=(sqlvars or {})), **kwargs}
262                result = pd.read_sql(stmt, connection, **pd_kwargs)
263            else:
264                # lazy returns a cursor
265                logger.debug(f'preparing a cursor for sql {sql} with parameters {sqlvars}')
266                result = connection.execute(stmt, **sqlvars)
267                keep = True
268            if not keep:
269                connection.close()
270            return result
271        return connection

273    def put(self, obj, name, sql=None, copy=False, append=True, chunksize=None,
274            transform=None, table=None, attributes=None, insert=False,
275            secrets=None, *args, **kwargs):
276        """ store sqlalchemy connection or insert data into an existing connection
277
278        Args:
279            obj (str|pd.DataFrame): the sqlalchemy connection string or a dataframe object
280            name (str): the name of the object
281            table (str): optional, if specified is stored along connection
282            sql (str): optional, if specified is stored along connection
283            copy (bool): optional, if True the connection is queried using sql
284                and the resulting data is stored instead, see below
285            attributes (dict): optional, set or update metadata.attributes
286
287        Returns:
288            metadata of the stored connection
289
290        Instead of inserting the connection specify ``copy=True`` to query data
291        and store it as a DataFrame dataset given by ``name``:
292
293        Args:
294            sql (str): sql to query
295            append (bool): if True the data is appended if exists already
296            chunksize (int): number of records to query in each chunk
297            transform (callable): passed as DataFrame.to_sql(method=)
298
299        Returns:
300            metadata of the inserted dataframe
301
302        To insert data via a previously stored connection, specify ``insert=True``:
303
304        Args:
305            insert (bool): specify True to insert via the connection
306            table (str): the table name to use for inserting data
307            append (bool): if False will replace any existing table, defaults to True
308            index (bool): if False will not attempt to create an index in target, defaults to False
309            chunksize (int): number of records to insert in each chunk
310
311        Returns:
312            metadata of the connection
313        """
314        meta = self.data_store.metadata(name)
315        if not insert and self._is_valid_url(obj):
316            # store a connection object
317            url = obj
318            cnx_name = name if not copy else '_cnx_{}'.format(name)
319            table = self._default_table(table or name)
320            metadata = self._put_as_connection(url, cnx_name, sql=sql, chunksize=chunksize,
321                                               table=table, attributes=attributes, **kwargs)
322            if copy:
323                secrets = self._get_secrets(metadata, secrets)
324                metadata = self._put_as_data(url, name, cnx_name,
325                                             sql=sql, chunksize=chunksize,
326                                             append=append, transform=transform,
327                                             secrets=secrets,
328                                             **kwargs)
329        elif meta is not None:
330            table = self._default_table(table or meta.kind_meta.get('table') or name)
331            metadata = self._put_via(obj, name, append=append, table=table, chunksize=chunksize,
332                                     transform=transform, **kwargs)
333        else:
334            raise ValueError('type {} is not supported by {}'.format(type(obj), self.KIND))
335        metadata.attributes.update(attributes) if attributes else None
336        return metadata.save()

338    def _put_via(self, obj, name, append=True, table=None, chunksize=None, transform=None,
339                 index_columns=None, index=True, **kwargs):
340        # write data back through the connection
341        # -- ensure we have a valid object
342        if not hasattr(obj, 'to_sql'):
343            warning('obj.to_sql() does not exist, trying pd.DataFrame(obj)')
344            obj = pd.DataFrame(obj)
345        # -- get the connection
346        connection = self.get(name, raw=True)
347        metadata = self.data_store.metadata(name)
348        if isinstance(obj, pd.DataFrame) and index:
349            index_cols = _dataframe_to_indexcols(obj, metadata, index_columns=index_columns)
350        else:
351            index_cols = index_columns
352        metadata.kind_meta['index_columns'] = index_cols
353        exists_action = 'append' if append else 'replace'
354        transform = transform
355        self._chunked_to_sql(obj, table, connection, chunksize=chunksize, method=transform,
356                             if_exists=exists_action, index=index, index_label=index_cols, **kwargs)
357        connection.close()
358        return metadata

399    def _get_connection(self, name, connection_str, secrets=None, keep=False):
400        from sqlalchemy import create_engine
401        # passwords should be encoded
402        # https://docs.sqlalchemy.org/en/13/core/engines.html#database-urls
403        encoded = lambda d: {
404            k: (quote_plus(v.decode('utf-8')) if isinstance(v, bytes)
405                else quote_plus(v)) for k, v in d.items() if isinstance(v, (str, bytes))
406        }
407        connection = None
408        cache_key = None
409        try:
410            # SECDEV: the cache key is a secret in order to avoid privilege escalation
411            # -- if it is not secret, user A could create the connection (=> cache)
412            # -- user B could reuse the connection by retrieving the dataset without secrets
413            # -- this way the user needs to have the same secrets in order to reuse the connection
414            enc_secrets = encoded(secrets or {})
415            connection_str = connection_str.format(**enc_secrets)
416            cache_key = sha256(f'{name}:{connection_str}'.encode('utf8')).hexdigest()
417            engine = self.__CNX_CACHE.get(cache_key) or create_engine(connection_str, **ENGINE_KWARGS)
418            connection = engine.connect()
419        except KeyError as e:
420            msg = ('{e}, ensure secrets are specified for connection '
421                   '>{connection_str}<'.format(**locals()))
422            raise KeyError(msg)
423        except Exception as e:
424            if connection is not None:
425                connection.close()
426                self.__CNX_CACHE.pop(cache_key, None)
427            raise
428        else:
429            if keep:
430                self.__CNX_CACHE[cache_key] = engine
431            else:
432                self.__CNX_CACHE.pop(cache_key, None)
433        return connection

435    def copy_from_sql(self, sql, connstr, name, chunksize=10000,
436                      append=False, transform=None, secrets=None, **kwargs):
437        connection = self._get_connection(name, connstr, secrets=secrets)
438        chunksize = chunksize or 10000  # avoid None
439        pditer = pd.read_sql(sql, connection, chunksize=chunksize, **kwargs)
440        with tqdm_if_interactive().tqdm(unit='rows') as pbar:
441            meta = self._chunked_insert(pditer, name, append=append,
442                                        transform=transform, pbar=pbar)
443        connection.close()
444        return meta

446    def _chunked_to_sql(self, df, table, connection, if_exists='append', chunksize=None, pbar=True, **kwargs):
447        # insert large df in chunks and with a progress bar
448        # from https://stackoverflow.com/a/39495229
449        chunksize = chunksize if chunksize is not None else 10000
450
451        def chunker(seq, size):
452            return (seq.iloc[pos:pos + size] for pos in range(0, len(seq), size))
453
454        def to_sql(df, table, connection, pbar=None):
455            for i, cdf in enumerate(chunker(df, chunksize)):
456                exists_action = if_exists if i == 0 else "append"
457                cdf.to_sql(table, con=connection, if_exists=exists_action, **kwargs)
458                if pbar:
459                    pbar.update(len(cdf))
460                else:
461                    print("writing chunk {}".format(i))
462
463        with tqdm_if_interactive().tqdm(total=len(df), unit='rows') as pbar:
464            to_sql(df, table, connection, pbar=pbar)

466    def _chunked_insert(self, pditer, name, append=True, transform=None, pbar=None):
467        # insert into om dataset
468        for i, df in enumerate(pditer):
469            if pbar is not None:
470                pbar.update(len(df))
471            should_append = (i > 0) or append
472            if transform:
473                df = transform(df)
474            try:
475                meta = self.data_store.put(df, name, append=should_append)
476            except Exception as e:
477                rows = df.iloc[0:10].to_dict()
478                raise ValueError("{e}: {rows}".format(**locals()))
479        return meta

494        secrets_specs = meta.kind_meta.get('secrets')
495        values = ({k:v for k, v in os.environ.items() if k.isupper() and isinstance(v, (str, bytes))}

507        # -- format secrets
508        if secrets:
509            secrets = _format_dict(secrets, **values, user=self._getuser())

524    def _sanitize_statement(self, sql, sqlvars, trusted=False):
525        # sanitize sql:string statement in two steps
526        # -- step 1: replace all {} variables by :notation
527        # -- step 2: replace all remaining {} variables from sqlvars
528        #    and issue a warning. step 2 is considered unsafe if
529        #    the sqlvars source cannot be trusted
530        # -- step 3: prepare a SQL statement with bound variables
531        # see https://realpython.com/prevent-python-sql-injection/#using-query-parameters-in-sql
532        if not isinstance(sql, str):
533            return sql
534        # replace all {...} variables with bound parameters
535        #    sql = "select * from foo where user={username}"
536        #       => "select * from foo where user=:username"
537        placeholders = list(string.Formatter().parse(sql))
538        vars = [spec[1] for spec in placeholders if spec[1]]
539        safe_replacements = {var: f':{var}' for var in vars}
540        sql = sql.format(**safe_replacements)
541        # build parameter list for tuples and lists
542        # -- sqlalchemy+pyodbc do not support lists of values
543        # -- list of values must be passed as single parameters
544        # -- e.g. sql=select * from x in :list
545        #    =>       select * from x in (:x_1, :x_2, :x_3, ...)
546        for k in vars:
547            # note we are iterating vars, not sqlvars
548            # -- sqlvars is not used in constructing sql text
549            v = sqlvars[k]
550            if isinstance(v, (list, tuple)):
551                bind_vars = {f'{k}_{i}': lv for i, lv in enumerate(v)}
552                placeholders = ','.join(f':{bk}' for bk in bind_vars)
553                sql = sql.replace(f':{k}', f'({placeholders})')
554                sqlvars.update(bind_vars)
555        try:
556            # format remaining {{}} for selection
557            #    sql = "select {{cols}} from foo where user=:username
558            #    =>    "select a, b from foo where user=:username
559            placeholders = list(string.Formatter().parse(sql))
560            vars = [spec[1] for spec in placeholders if spec[1]]
561            if vars and trusted != self.sign(sqlvars):
562                warnings.warn(f'Statement >{sql}< contains unsafe variables {vars}. Use :notation or sanitize input.')
563            sql = sql.format(**{**sqlvars, **safe_replacements})
564        except KeyError as e:
565            raise KeyError('{e}, specify sqlvars= to build query >{sql}<'.format(**locals()))
566        # prepare sql statement with bound variables
567        try:
568            stmt = sqlalchemy.sql.text(sql)
569        except StatementError as exc:
570            raise
571        return stmt

574def _is_valid_url(url):
575    # check if we have a valid url with a registered backend
576    import sqlalchemy
577
578    try:
579        url = sqlalchemy.engine.url.make_url(url)
580        drivername = url.drivername.split('+')[0]  # e.g. mssql+pyodbc => mssql
581        valid = url.drivername in sqlalchemy.dialects.__all__
582        valid |= sqlalchemy.dialects.registry.load(drivername) is not None
583    except:
584        valid = False
585    return valid

588def _dataframe_to_indexcols(df, metadata, index_columns=None):
589    # from a dataframe get index column names
590    # works like pd.DataFrame.to_sql except for creating default index_i cols
591    # for any missing (None) index labels in a MultiIndex.
592    index_cols = metadata.kind_meta.get('index_columns') or index_columns or list(df.index.names)
593    multi = isinstance(df.index, pd.MultiIndex)
594    if index_cols is not None:
595        for i, col in enumerate(index_cols):
596            if col is None:
597                index_cols[i] = 'index' if not multi else 'index_{}'.format(i)
598    return index_cols

611def _format_dict(d, replace=None, **kwargs):
612    for k, v in dict(d).items():
613        if replace:
614            del d[k]
615            k = k.replace(*replace) if replace else k
616        d[k] = v.format_map(KeepMissing(kwargs)) if isinstance(v, str) else v

620def load_sql(om=None, kind=SQLAlchemyBackend.KIND):
621    """
622    load ipython sql magic, loading all sql alchemy connections
623
624    Usage:
625        !pip install ipython-sql
626
627        # prepare some connection, insert some data
628        df = pd.DataFrame(...)
629        om.datasets.put('sqlite:///test.db', 'testdb')
630        om.datasets.put(df, 'testdb', table='foobar', insert=True)
631
632        from omegaml.backends.sqlalchemy import load_sql
633        load_sql()
634
635        # list registered connections
636        %sql
637            omsql://testdb
638
639        # run queries
640        %sql omsql://testdb select * from foobar
641
642    See Also:
643        https://pypi.org/project/ipython-sql/
644
645    Args:
646        om (Omega): optional, specify omega instance, defaults to om.setup()
647        kind (str): the backend's kind, used to find connections to register
648           with the sql magic
649
650    Returns:
651        None
652    """
653    from unittest.mock import MagicMock
654    from IPython import get_ipython
655    import omegaml as om
656    from sql.connection import Connection # noqa
657
658    class ConnectionShim:
659        # this is required to trick sql magic into accepting existing connection objects
660        # (by default sql magic expects connection strings, not connection objects)
661        def __init__(self, url, conn):
662            self.session = conn
663            self.metadata = MagicMock()
664            self.metadata.bind.url = url
665            self.dialect = getattr(conn, 'dialect', 'omsql')
666
667    # load sql magic
668    ipython = get_ipython()
669    ipython.magic('load_ext sql')
670    # load registered sqlalchemy datasets
671    om = om or om.setup()
672    for ds in om.datasets.list(kind=kind, raw=True):
673        cnxstr = 'omsql://{ds.name}'.format(**locals())
674        conn = om.datasets.get(ds.name, raw=True)
675        Connection.connections[cnxstr] = ConnectionShim(cnxstr, conn)
676    ipython.magic('sql')
